How it works

How it works

  1. Read the current time with new Date().
  2. Format hours, minutes, and seconds with leading zeros.
  3. Render into the <time> element and update its datetime attribute.
  4. Schedule the next tick exactly at the next second boundary to avoid drift.

Core logic

// Pads a number to 2 digits: 7 → "07"
const pad2 = (n) => String(n).padStart(2, "0");

// Renders current time as HH:MM:SS (24-hour)
function render(now = new Date()) {
  const h = pad2(now.getHours());
  const m = pad2(now.getMinutes());
  const s = pad2(now.getSeconds());

  hh.textContent = h;
  mm.textContent = m;
  ss.textContent = s;

  // Machine-readable time for assistive tech and SEO
  clockEl.setAttribute("datetime", now.toISOString());
}

// Aligns updates to the next exact second to prevent drift
function scheduleNextTick() {
  const now = new Date();
  render(now);
  const msUntilNextSecond = 1000 - now.getMilliseconds();
  setTimeout(scheduleNextTick, msUntilNextSecond);
}
